📊 Google 試算表同步設定
設定步驟：
1. 開啟 Google Apps Script → 建立新專案
2. 貼入下方程式碼並部署為「網頁應用程式」（執行身分：我、存取：所有人）
3. 複製部署 URL 貼入下方欄位
// ★ 複製以下全部內容貼入 Apps Script，並將 YOUR_SPREADSHEET_ID 換成試算表ID ★ function processData(data) { var ss = SpreadsheetApp.openById('YOUR_SPREADSHEET_ID'); var type = data.type || 'order'; if (type === 'ping') return {ok: true, message: 'connected'}; if (type === 'order') { var s = ss.getSheetByName('訂單') || ss.insertSheet('訂單'); if (s.getLastRow() === 0) s.appendRow(['訂單ID','日期','姓名','付款銀行','後五碼','寄送','門市','品項','小計','運費','折扣','總計','Email','狀態']); var items = (data.items||[]).map(function(i){ return i.name+(i.spec?'('+i.spec+')':'')+'×'+i.qty+'=$'+i.unitPrice; }).join('；'); // Check if order already exists (avoid duplicates) var finder = s.createTextFinder(data.id).findNext(); if (!finder) { s.appendRow([data.id,data.date,data.name,data.bank||'',data.code||'', data.shipping,data.store||'',items,data.subtotal,data.shippingFee, data.couponDiscount||0,data.total,data.email,data.status]); } } if (type === 'product') { var sp = ss.getSheetByName('商品') || ss.insertSheet('商品'); if (sp.getLastRow() === 0) sp.appendRow(['名稱','分類','子分類','價格','最高價','規格','顏色','尺寸','標籤','描述']); var fp = sp.createTextFinder(data.name).findNext(); if (fp) { sp.getRange(fp.getRow(), 1, 1, 10).setValues([[data.name,data.cat,data.subCat, data.price,data.priceMax,data.specs,data.colors||'',data.sizes||'',data.badge,data.desc]]); } else { sp.appendRow([data.name,data.cat,data.subCat,data.price,data.priceMax, data.specs,data.colors||'',data.sizes||'',data.badge,data.desc]); } } if (type === 'member') { var sm = ss.getSheetByName('會員') || ss.insertSheet('會員'); if (sm.getLastRow() === 0) sm.appendRow(['Email','姓名','電話','生日','性別']); var fm = sm.createTextFinder(data.email).findNext(); if (fm) { sm.getRange(fm.getRow(), 1, 1, 5).setValues([[data.email,data.name,data.phone,data.birth,data.gender]]); } else { sm.appendRow([data.email,data.name,data.phone,data.birth,data.gender]); } } return {ok: true}; } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var result = processData(data); return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { // Handle payload sent via GET (for CORS workaround) if (e.parameter && e.parameter.payload) { var data = JSON.parse(decodeURIComponent(e.parameter.payload)); var result = processData(data); return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // Normal GET: read sheet data var ss = SpreadsheetApp.openById('YOUR_SPREADSHEET_ID'); var type = (e.parameter && e.parameter.type) || 'orders'; var sheetName = type==='orders'?'訂單':type==='products'?'商品':'會員'; var sheet = ss.getSheetByName(sheetName); if (!sheet) return ContentService.createTextOutput('[]').setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify(sheet.getDataRange().getValues())) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } }
